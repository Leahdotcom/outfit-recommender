  <input type="text" id="city" placeholder="Enter city name">

  <button onclick="getOutfit()">Get My Outfit</button>

  <div id="result"></div>


  <script>

    const backendUrl = "https://script.google.com/macros/s/AKfycbxUWIK6JULruoE-4N9o057vt44DGdxf7PEG_x_4w9Dfgx18WBFfy0wpkyzHGUD3MJfirA/exec";


    function getOutfit() {

      const city = document.getElementById("city").value;

      document.getElementById("result").innerHTML = "⏳ Getting your outfit...";


      fetch(backendUrl, {

        method: 'POST',

        body: JSON.stringify({ city: city }),

        headers: {

          'Content-Type': 'application/json'

        }

      })

      .then(res => res.json())

      .then(response => {

        const links = (response.images || []).map(url =>

          `<a href="${url}" target="_blank">📷 View Outfit Image</a>`

        ).join("");

        document.getElementById("result").innerHTML =

          `<p>${response.text}</p>${links}`;

      })

      .catch(err => {

        document.getElementById("result").innerHTML = "❌ Error loading outfit.";

        console.error(err);

      });

    }

  </script>
